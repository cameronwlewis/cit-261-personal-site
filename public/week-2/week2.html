Here's a function, with the parameter and some conditional statements labeled:

function funkyFunction(disco /* <---'disco' is now our parameter */ ) {
        // use regex to test my "disco" variable
        specific condition
        var regexPattern = /D?d?ance/;

        // These are some conditional statements
        if (regexPattern.test(disco))
            document.getElementById("dancing-status").innerHTML =
            "Let's groove!";
        else
            document.getElementById("dancing-status").innerHTML =
            "Guess we ain't dancin today...";
    }
