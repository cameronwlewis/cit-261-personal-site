Here's (a summary of) the code showing an implementation of the globe & rocket CSS animation below.

HTML:

<div id="rocket-globe-container">
    <svg id="rocket-globe" viewBox="0 0 1500 3500" xmlns="http://www.w3.org/2000/svg"/>
        <g class="globe"/>
        <g id="rocket-sitting" visibility="visible"/>
        <g id="rocket-blasting" visibility="hidden"/>
    </svg>
</div>

CSS:

#rocket-globe{
    height: auto;
    display: block;
    z-index: 1;
    margin-top: -30%;
    margin-left: auto;
    margin-right: auto;
    overflow: auto;
}

@media screen and (max-width: 2000px) {

        max-width: 30%;
    }

        max-width: 40%;
    }
}

    #rocket-globe {
        max-width: 50%;
    }
}

@media screen and (max-width: 700px) {
    #rocket-globe {
        max-width: 60%;
    }

@media screen and (max-width: 350px) {
    #rocket-globe {
        max-width: 70%;
    }

.blast-off {
    animation: blastOff 2s;
}

@keyframes blastOff {
   from {transform: translate(0px, 0px)}
   to {transform: translate(0px, -3000px)}
    40% {opacity: 1}
    50% {opacity: 0}
}

JavaScript:

var rocket_sitting = document.getElementById("rocket-sitting");
var rocket_blasting = document.getElementById("rocket-blasting");

function showFlames() {
    rocket_sitting.style.visibility = "hidden";
    rocket_blasting.style.visibility = "visible";
    rocket_blasting.classList.add("blast-off");
}

function hideFlames() {
    rocket_sitting.style.visibility = "visible";
    rocket_blasting.style.visibility = "hidden";
    rocket_blasting.classList.remove("blast-off");
}
//document.getElementById('rocket-sitting').addEventListener('touchstart',listener, false);
rocket_sitting.addEventListener("click", function () {
        showFlames();
    }
);

rocket_blasting.addEventListener("animationend", listener, false);
rocket_blasting.addEventListener("animationstart", listener, false);

function listener(e) {
    switch(e.type) {
        case "animationstart":
            showFlames();
            break;
        case "animationend":
            hideFlames();
            break;
    }
}
